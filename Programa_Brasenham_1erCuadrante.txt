#include<stdio.h>
#include<string.h>
#include<math.h>

int main(){
	
	int x1, y1, x2, y2;
	
	printf("Escribe la primera coordenada de x: ");
	scanf("%i", &x1);
	printf("Escribe la primera coordenada de y: ");
	scanf("%i", &y1);
	printf("Escribe la segunda coordenada de x: ");
	scanf("%i", &x2);
	printf("Escribe la segunda coordenada de y: ");
	scanf("%i", &y2);
	
	int dx = x2 - x1;
	int dy = y2 - y1;
	int dx2 = 2 * dx;
	int dy2 = 2 * dy;
	
	if(dy == 0 ){	
		printf("Este trazo de linea es completamente horizontal \n");
	}
	if(dx == 0){
		printf("Este trazo de linea es completamente vertical \n");
	}
	

	if (dx >= dy){
		int c = dy2 - dx2;
		int pk =  dy2 - dx;
		printf("Los puntos iniciales son: %i, %i \nY el trazado de la linea sera por los puntos: \n(X, Y) \n", x1, y1);
		while(x1 < x2 || y1 < y2){
			if (pk < 0){
				pk += dy2;
				x1 += 1;
				printf(" %i, %i \n", x1, y1);
			}
			else{
				pk += c;
				x1 += 1;
				y1 += 1;
				printf(" %i, %i \n", x1, y1);
			}			
		}			
	}
	
	if (dy > dx){
		int c = dx2 - dy2;
		int pk = dx2 - dy;
		printf("Los puntos iniciales son: %i, %i \nY el trazado de la linea sera por los puntos: \n(X, Y) \n", x1, y1);
		while(x1 < x2 || y1 < y2){
			if(pk < 0){
				pk += dx2;
				y1 += 1;
				printf(" %i, %i \n", x1, y1);	
			}
			else{
				pk += c;
				x1 += 1;
				y1 += 1;
				printf(" %i, %i \n", x1, y1);
			}
		}
		
	}	
	return 0;
}
